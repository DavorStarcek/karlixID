  Microsoft.EntityFrameworkCore.SqlServer `
  -Context ScaffoldTempContext `
  -ContextDir "Data\ScaffoldCtx" `
  -OutputDir "Models\Scaffold" `
  -NoOnConfiguring `
  -UseDatabaseNames `
  -Force `
  -ExcludeTables AspNetUsers,AspNetRoles,AspNetUserClaims,AspNetRoleClaims,AspNetUserLogins,AspNetUserTokens,AspNetUserRoles
